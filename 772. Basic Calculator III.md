```java
import java.util.Stack;

class Solution {
    public int calculate(String s) {
        Stack<Long> nums = new Stack<>();
        Stack<Character> ops = new Stack<>();

        for (int i = 0; i < s.length(); i++) {
            char c = s.charAt(i);
            if (Character.isDigit(c)) {
                int j = i + 1;
                while (j < s.length() && Character.isDigit(s.charAt(j))) {
                    j++;
                }
                nums.push(Long.parseLong(s.substring(i, j)));
                i = j - 1;
            } else if (c == '(') {
                ops.push(c);
            } else if (c == ')') {
                while (!ops.isEmpty() && ops.peek() != '(') {
                    long tmp = nums.pop();
                    nums.push(operate(ops.pop(), nums.pop(), tmp));
                }
                // pop (
                ops.pop();
            } else {
                //operand: +-*/
                // bug 2*(5+
                while (!ops.isEmpty() && ops.peek() != '(' && !currHigher(c, ops.peek())) {
                    long tmp = nums.pop();
                    nums.push(operate(ops.pop(), nums.pop(), tmp));
                }
                ops.push(c);
            }
        }

        long res = nums.pop();
        while (!ops.isEmpty()) {
            res = operate(ops.pop(), nums.pop(), res);
        }

        return (int)res;
    }

    private long operate(char operate, long num1, long num2) {
        switch (operate) {
            case '+':
                return num1 + num2;
            case '-':
                return num1 - num2;
            case '*':
                return num1 * num2;
            case '/':
                return num1 / num2;
            default:
                return 0;
        }
    }

    private boolean currHigher(char curr, char stackTop) {
        return (curr == '*' || curr == '/')
                && (stackTop == '+' || stackTop == '-');
    }
}
```

```Java
                ops.push(c);
            } else if (c == ')') {
                //calculate until (
                long num = nums.pop();
                while (!ops.isEmpty() && ops.peek() != '(') {
                    num = operate(nums.pop(), num, ops.pop());
                }
                ops.pop();
                nums.push(num);
            } else if (c != ' ') {
                // + - * /
                long num = nums.pop();
                // if op in stack has higher or equal priority, calculate previous nums
                while (!ops.isEmpty() && higherPriority(ops.peek(), c) && ops.peek() != '(' ) {
                    num = operate(nums.pop(), num, ops.pop());
                }
                nums.push(num);
                ops.push(c);
            }

        }
        long res = nums.pop();
        while (!ops.isEmpty()) {
            res = operate(nums.pop(), res, ops.pop());
        }
        return (int)res;
    }

    // op1 has higher priority than op2
    private boolean higherPriority(char op1, char op2) {
        if (op1 == '*' || op1 == '/') {
            return true;
        }
        return op2 == '+' || op2 == '-';
    }

            case '+':
                return num1 + num2;

                return num1 - num2;
```

```java
import java.util.*;

class Solution {
    public int calculate(String s) {
        if (s == null || s.length() == 0) return 0;
        Stack<Long> nums = new Stack<>();
        Stack<Character> ops = new Stack<>();

        for (int i = 0; i < s.length(); i++) {
            if (s.charAt(i) == ' ') continue;
            if (Character.isDigit(s.charAt(i))) {
                int end = i;
                while (end < s.length() && Character.isDigit(s.charAt(end))) {
                    end++;
                }
                Long num = Long.valueOf(s.substring(i, end));
                nums.push(num);
                //i++ later
                i = end - 1;
            } else {
                char c = s.charAt(i);
                if (c == '(') {
                    ops.push(c);
                } else if (c == ')') {
                    while (ops.peek() != '(') {
                        nums.push(operate(ops.pop(), nums.pop(), nums.pop()));
                    }
                    ops.pop();
                } else if (c == '+' || c == '-' || c == '*' || c == '/'){
                    while (!ops.isEmpty() && !currHigherPriority(c, ops.peek())) {
                        nums.push(operate(ops.pop(), nums.pop(), nums.pop()));
                    }
                    ops.push(c);
                }

            }
        }
        while (!ops.isEmpty()) {
            nums.push(operate(ops.pop(), nums.pop(), nums.pop()));
        }
        return Math.toIntExact(nums.pop());
    }
    private long operate(char op, long num2, long num1) {
        switch (op) {
            case '*': {
                return num1 * num2;
            }
            case '/': {
                return num1 / num2;
            }
            case '+': {
                return num1 + num2;
            }
            case '-': {
                return num1 - num2;
            }
            default: {
                return 0;
            }
        }
    }
    private boolean currHigherPriority(char op1, char op2) {
        //stack has no ')'
        if (op2 == '(' ) {
            return true;
        }
        if (op1 == '*' || op1 == '/') {
            return op2 == '+' || op2 == '-';
        } else {
            return false;
        }
    }
}
```

```java
import java.util.*;

class Solution {
    public static int calculate(String s) {
        if (s == null || s.length() == 0) return 0;
        Stack<Integer> nums = new Stack<>(); // the stack that stores numbers
        Stack<Character> ops = new Stack<>(); // the stack that stores operators (including parentheses)
        int num = 0;
        for (int i = 0; i < s.length(); i++) {
            char c = s.charAt(i);
            if (c == ' ') continue;
            if (Character.isDigit(c)) {
                num = c - '0';
                // iteratively calculate each number
                while (i < s.length() - 1 && Character.isDigit(s.charAt(i+1))) {
                    num = num * 10 + (s.charAt(i+1) - '0');
                    i++;
                }
                nums.push(num);
                num = 0; // reset the number to 0 before next calculation
            } else if (c == '(') {
                ops.push(c);
            } else if (c == ')') {
                // do the math when we encounter a ')' until '('
                while (ops.peek() != '(') {
                    nums.push(operation(ops.pop(), nums.pop(), nums.pop()));
                }
                ops.pop(); // get rid of '(' in the ops stack
            } else if (c == '+' || c == '-' || c == '*' || c == '/') {
                while (!ops.isEmpty() && precedence(c, ops.peek())) {
                    nums.push(operation(ops.pop(), nums.pop(), nums.pop()));
                }
                ops.push(c);
            }
        }
        while (!ops.isEmpty()) {
            nums.push(operation(ops.pop(), nums.pop(), nums.pop()));
        }
        return nums.pop();
    }

    private static int operation(char op, int b, int a) {
        switch (op) {
            case '+': return a + b;
            case '-': return a - b;
            case '*': return a * b;
            case '/': return a / b; // assume b is not 0
        }
        return 0;
    }
    // helper function to check precedence of current operator and the uppermost operator in the ops stack
    private static boolean precedence(char op1, char op2) {
        if (op2 == '(' || op2 == ')') return false;
        if ((op1 == '*' || op1 == '/') && (op2 == '+' || op2 == '-')) return false;
        return true;
    }

}
```

因为用的stack存数字，计算的时候会是先计算后面的，这样的话就得解决*+这样顺序的式子。遇到')'也一样，因为括号优先级高，得先计算了。如果是同级的符号，也得先计算了，如例子2。存integer在stack里的话，有一些trick在里面，例如MAX_VALUE + 1 = MIN_VALUE,正好是对的。
