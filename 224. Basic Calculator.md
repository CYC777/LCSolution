class Solution {
    public int calculate(String s) {
        Stack<Long> nums = new Stack<>();
        Stack<Character> ops = new Stack<>();
        nums.push(0L);
        for (int i = 0; i < s.length(); i++) {
            if (Character.isDigit(s.charAt(i))) {
                int j = i;
                while (j < s.length() && Character.isDigit(s.charAt(j))) {
                    j++;
                }
                Long num = Long.parseLong(s.substring(i, j));
                nums.push(num);
                i = j - 1;
            } else if (s.charAt(i) == ')'){
                long num = nums.pop();
                while (!ops.isEmpty() && ops.peek() != '(') {
                    num = operate(nums.pop(), num, ops.pop());
                }
                ops.pop();
                nums.push(num);

            } else if (s.charAt(i) == '(') {
                ops.push(s.charAt(i));
            } else if (s.charAt(i) != ' ') {
                // + or -
                if (i > 0 && s.charAt(i - 1) == '(') {
                    nums.push(0L);
                }
                long num = nums.pop();
                while (!ops.isEmpty() && ops.peek() != '(') {
                    num = operate(nums.pop(), num, ops.pop());
                }
                nums.push(num);
                ops.push(s.charAt(i));
            }
        }
        long res = nums.pop();

        while (!ops.isEmpty()) {
            res = operate(nums.pop(), res, ops.pop());
        }
        return (int)res;
    }

    private long operate(long num1, long num2, char op) {
        switch (op) {
            case '+':
                return num1 + num2;
            case '-':
                return num1 - num2;
            default:
                return 0;
        }
    }
}

import java.util.Stack;

class Solution {
    public int calculate(String s) {
        Stack<Long> nums = new Stack<>();
        Stack<Character> ops = new Stack<>();

        for (int i = 0; i < s.length(); i++) {
            char c = s.charAt(i);
            if (Character.isDigit(c)) {
                int j = i + 1;
                while (j < s.length() && Character.isDigit(s.charAt(j))) {
                    j++;
                }
                long currNum = Long.parseLong(s.substring(i, j));
                nums.push(currNum);
                //bug
                i = j - 1;
            } else if (c == '+' || c == '-') {
                int j = i;
                while (j >= 0 && !Character.isDigit(s.charAt(j)) && s.charAt(j) != '(') {
                    j--;
                }
                if (j < 0 || s.charAt(j) == '(') {
                    nums.push(0L);
                }

                while (!ops.isEmpty() && ops.peek() != '(' && ops.peek() != ')') {
                    long tmp = nums.pop();
                    long res = operate(ops.pop(), nums.pop(), tmp);
                    nums.push(res);
                }

                ops.push(c);
            } else if (c == '(') {
                ops.push(c);
            } else if (c == ')') {
                while (!ops.isEmpty() && ops.peek() != '(') {
                    long tmp = nums.pop();
                    long res = operate(ops.pop(), nums.pop(), tmp);
                    nums.push(res);
                }
                ops.pop();
            }
        }

        long res = nums.pop();
        while (!ops.isEmpty()) {
            res = operate(ops.pop(), nums.pop(), res);
        }

        return (int)res;
    }

    private long operate(char op, long num1, long num2) {
        switch (op) {
            case '+':
                return num1 + num2;
            case '-':
                return num1 - num2;
            default:
                return 0;
        }
    }
}
