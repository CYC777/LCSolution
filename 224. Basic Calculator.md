class Solution {
    public int calculate(String s) {
        Stack<Long> nums = new Stack<>();
        Stack<Character> ops = new Stack<>();
        nums.push(0L);
        for (int i = 0; i < s.length(); i++) {
            if (Character.isDigit(s.charAt(i))) {
                int j = i;
                while (j < s.length() && Character.isDigit(s.charAt(j))) {
                    j++;
                }
                Long num = Long.parseLong(s.substring(i, j));
                nums.push(num);
                i = j - 1;
            } else if (s.charAt(i) == ')'){
                long num = nums.pop();
                while (!ops.isEmpty() && ops.peek() != '(') {
                    num = operate(nums.pop(), num, ops.pop());
                }
                ops.pop();
                nums.push(num);

            } else if (s.charAt(i) == '(') {
                ops.push(s.charAt(i));
            } else if (s.charAt(i) != ' ') {
                // + or -
                if (i > 0 && s.charAt(i - 1) == '(') {
                    nums.push(0L);
                }
                long num = nums.pop();
                while (!ops.isEmpty() && ops.peek() != '(') {
                    num = operate(nums.pop(), num, ops.pop());
                }
                nums.push(num);
                ops.push(s.charAt(i));
            }
        }
        long res = nums.pop();

        while (!ops.isEmpty()) {
            res = operate(nums.pop(), res, ops.pop());
        }
        return (int)res;
    }

    private long operate(long num1, long num2, char op) {
        switch (op) {
            case '+':
                return num1 + num2;
            case '-':
                return num1 - num2;
            default:
                return 0;
        }
    }
}

//遇到+-的时候计算，而不是数字 (7)-(0)+(4)
    // 先push 0， 防止-开头，因为最后判断 ops.isEmpty, nums多了个0会被忽略

//算法：
//    空格 : 跳过
//    ( : 直接加入 ops 中，等待与之匹配的 )
//    ) : 使用现有的 nums 和 ops 进行计算，直到遇到左边最近的一个左括号为止，计算结果放到 nums
//    数字 : 从当前位置开始继续往后取，将整一个连续数字整体取出，加入 nums
//    + - * / ^ % : 需要将操作放入 ops 中。在放入之前先把栈内可以算的都算掉（只有「栈内运算符」比「当前运算符」优先级高/同等，才进行运算），使用现有的 nums 和 ops 进行计算，直到没有操作或者遇到左括号，计算结果放到 nums

//细节： operate哪个数在前面，用long
