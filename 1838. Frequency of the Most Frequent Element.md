class Solution {
    public int maxFrequency(int[] nums, int k) {
        Arrays.sort(nums);
        //k + sum >= (j - i + 1) * nums[j]
        int i = 0, j = 0;
        long sum = 0;
        int res = 0;
        while (j < nums.length) {
            sum += nums[j];
            while (k + sum < (j - i + 1) * (long)nums[j]) {
                sum -= nums[i];
                i++;
            }
            res = Math.max(res, j - i + 1);
            j++;
        }
        return res;
    }
}
